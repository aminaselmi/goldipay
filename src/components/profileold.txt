import { useSelector, useDispatch } from "react-redux";
import { useEffect } from "react";
import FilterData from "../Filter/FilterData";
import { Flex, Box, Spacer } from "@chakra-ui/react";
import { useMediaQuery } from "@chakra-ui/react";
import Loading from "../components/Loading";
import localProducts from "../data/products";

const Products = () => {
  const dispatch = useDispatch();
  const products = useSelector((store) => store?.dataReducer?.products);
  const loading = useSelector((store) => store?.dataReducer?.isLoading);
  const [isLargerThan] = useMediaQuery("(min-width: 768px)");

  useEffect(() => {
    dispatch({
      type: "SET_LOCAL_PRODUCTS",
      payload: localProducts,
    });
  }, [dispatch]);

  return (
    <div className="AllProducts">
      <br />
      {loading ? (
        <Loading />
      ) : (
        <Flex flexDirection={isLargerThan ? "row" : "column"}>
          <Box w={isLargerThan ? "15%" : "100%"}>
            <FilterData />
          </Box>

          <Spacer />

          <Box w={isLargerThan ? "80%" : "100%"}>
            {products?.map((item) => (
              <Box key={item.id} border="1px solid gray" p="10px" m="10px">
                <img src={item.images[0]} width="150px" alt={item.name} />
                <h3>{item.name}</h3>
                <p>₹{item.final_price}</p>
              </Box>
            ))}
          </Box>
        </Flex>
      )}
    </div>
  );
};

export default Products;
